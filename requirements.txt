# OpenCV on headless Linux servers (Render, Railway, EC2):
#   Use: opencv-python-headless instead of opencv-python
#   pip install opencv-python-headless
#
# rembg model download (U2Net, ~170MB):
#   Downloaded automatically on first use from HuggingFace Hub
#   Set REMBG_HOME env var to cache in a persistent location:
#   export REMBG_HOME=/path/to/model/cache
#
# OpenCV on macOS (if cv2 import fails):
#   brew install opencv
#   pip install opencv-python
#
# OpenCV on Ubuntu/Debian (libGL error):
#   sudo apt-get install libgl1-mesa-glx libglib2.0-0 libsm6 libxext6 libxrender-dev
#
